from multiprocessing import Process, Queue
import os, time, random

def read(q):
    value = q.get(True)
    print('读取的内容是：%s'%value)
    q.put('这是read函数写入的内容')
    
def write(q):
    q.put('我是队列写入的内容')
    value = q.get(True)
    print(value)

if __name__ == '__main__':
    q = Queue()
    pw = Process(target=write,args=(q,))
    pr = Process(target=read,args=(q,))
    pw.start()
    pr.start()
    pw.join()
    pr.join()
    print('所有程序运行完毕')
